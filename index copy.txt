  <script>
    // year
    document.getElementById('yr').textContent = new Date().getFullYear();

    // calculator logic
    const form = document.getElementById('calcForm');
    const resultsEl = document.getElementById('results');
    const caloriesOut = document.getElementById('caloriesOut');
    const proteinOut  = document.getElementById('proteinOut');
    const resetBtn = document.getElementById('resetBtn');

    function round(n){ return Math.round(n); }

    form.addEventListener('submit', function(e){
      e.preventDefault();

      const sex = document.getElementById('sex').value;
      const age = parseFloat(document.getElementById('age').value);
      const heightIn = parseFloat(document.getElementById('height').value);
      const weightLb = parseFloat(document.getElementById('weight').value);
      const activity = parseFloat(document.getElementById('activity').value);
      const goal = document.getElementById('goal').value;

      if(!sex || !age || !heightIn || !weightLb || !activity || !goal){
        alert('Please complete all required fields.');
        return;
      }

      const weightKg = weightLb * 0.45359237;
      const heightCm = heightIn * 2.54;
      let bmr;
      if(sex === 'female'){
        bmr = 10*weightKg + 6.25*heightCm - 5*age - 161;
      } else {
        bmr = 10*weightKg + 6.25*heightCm - 5*age + 5;
      }

      const tdee = bmr * activity;
      let calories = tdee;
      if(goal === 'fatloss'){ calories = tdee * 0.8; }
      if(goal === 'maintain'){ calories = tdee; }
      if(goal === 'muscle'){ calories = tdee * 1.1; }

      let pRate = 0.9;
      if(goal === 'fatloss'){ pRate = 1.0; }
      if(goal === 'maintain'){ pRate = 0.9; }
      if(goal === 'muscle'){ pRate = 1.0; }
      const protein = weightLb * pRate;

      caloriesOut.textContent = round(calories).toLocaleString();
      proteinOut.textContent  = round(protein) + ' g';
      resultsEl.style.display = 'block';
    });

    resetBtn.addEventListener('click', function(){
      form.reset();
      resultsEl.style.display = 'none';
      caloriesOut.textContent = '—';
      proteinOut.textContent  = '—';
    });
  </script>
